%% --- DEFINING POSTER COLORS --- %%
\newcommand{\SetPrimaryColor}[1]{\def\PrimaryColor{#1}}
\newcommand{\SetSecondaryColor}[1]{\def\SecondaryColor{#1}}
\newcommand{\SetAccentColor}[1]{\def\AccentColor{#1}}
\newcommand{\SetThemeColor}[1]{\def\ThemeColor{#1}}
\newcommand{\SetColorTitleMainBox}[1]{\def\ColorTitleMainBox{#1}}
\newcommand{\SetColorTitleHighlightBox}[1]{\def\ColorTitleHighlightBox{#1}}
\newcommand{\SetColorTitleGrayBox}[1]{\def\ColorTitleGrayBox{#1}}
\newcommand{\SetColorTitleDarkBox}[1]{\def\ColorTitleDarkBox{#1}}
\newcommand{\WhichPalette}[1]{\def\Palette{#1}}

\newcommand{\SetUFVPalette}{
    % Poster Colors
    \SetPrimaryColor{ufvRed}
    \SetSecondaryColor{ufvYellow}
    \SetAccentColor{ufvGrey}
    \SetThemeColor{ufvRed}
    % Color Boxes Title Color
    \SetColorTitleMainBox{black}
    \SetColorTitleHighlightBox{white}
    \SetColorTitleGrayBox{white}
    \SetColorTitleDarkBox{white}
}

\newcommand{\SetPalette}[1]{%
    \IfEqCase{#1}{%
        % --- CASE: GISC ---
        {gisc}{
        \WhichPalette{gisc}
        \SetPrimaryColor{giscBlue}
        \SetSecondaryColor{giscLightBlue}
        \SetAccentColor{giscGray}
        \SetThemeColor{giscRed}
        \SetColorTitleMainBox{black}
        \SetColorTitleHighlightBox{white}
        \SetColorTitleGrayBox{white}
        \SetColorTitleDarkBox{black}}
        % --- CASE: FANCY ---
        {fancy}{
        \WhichPalette{fancy}
        \SetPrimaryColor{fancyBlue}
        \SetSecondaryColor{fancyLightBlue}
        \SetAccentColor{fancyGray}
        \SetThemeColor{fancyRed}
        \SetColorTitleMainBox{black}
        \SetColorTitleHighlightBox{white}
        \SetColorTitleGrayBox{white}
        \SetColorTitleDarkBox{white}}
        [ 
        % Default case   
        %\PackageWarning{CleanTeX-Poster}{Palette '#1' not found. Falling back to 'ufv'.}
        \WhichPalette{ufv}
        \SetUFVPalette]
}}

\SetUFVPalette
