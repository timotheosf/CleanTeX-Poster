% --- COMANDOS PERSONALIZADOS ---
\newcommand{\sectionTitle}[1]{%
  \vspace{1.5em}
  \begin{tcolorbox}[
      enhanced,
      sharp corners,
      frame hidden,
      boxrule=0pt,
      colback=\PrimaryColor!5!white,
      borderline west={5pt}{0pt}{\PrimaryColor},
      left=0.5em,
      top=0.4em,
      bottom=0.4em,
      halign=center,
      valign=center
  ]
      {\setSectionFont\bfseries\color{\PrimaryColor} #1}
  \end{tcolorbox}
  \vspace{0.5em}
}
