%> Show color in box
\makeatletter
\def\parseRGB#1,#2,#3\@nil{%
    \def\colorR{#1}%
    \def\colorG{#2}%
    \def\colorB{#3}%
}
\makeatletter
%> Print color in a square
\newcommand{\ShowColor}[3][white]{%
    \convertcolorspec{named}{#2}{RGB}{\currentRGB}%
    \expandafter\parseRGB\currentRGB\@nil
    \begin{tikzpicture}
        % Desenha um quadrado exato do tamanho da minipage
        \fill[#2] (0,0) rectangle (\linewidth,\linewidth);
        % O texto fica ancorado pertinho da margem inferior esquerda
        \node[text=#1, anchor=south west, align=left, font=\sffamily\Large] at (0.2,0.2) {%
            \textbf{#3}\\[0.2em]R: \colorR\\G: \colorG\\B: \colorB%
        };
    \end{tikzpicture}%
    \makeatother%
}
